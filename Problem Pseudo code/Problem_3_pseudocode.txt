List sequenceOne = file.readline !
List sequenceTwo = file.readline !

List outputSequence !
outputSequence ++ 0 !
i = 0 !

while ( i < (length(sequenceTwo) div 2) -1 ) {
outputSequence ++ sequenceTwo index (i*2 + 1) !

}

OUTPUT ( outputSequence ) !
