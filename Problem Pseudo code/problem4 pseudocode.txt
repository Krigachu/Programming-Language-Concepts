--- important notes ---

-- 1 stream input
-- That is at least three values are taken from the single input sequence
-- The third of these is output first
-- The second is doubled and then output
-- The first value is trebled decremented and output.
-- This sequence repeats.


--- pseudocode ---

List stream = file readline !
List outputSequence !
i = 0 !

if ( length (stream) > 2) then {
    while ( length (stream) - i > 2 ) then {
        outputSequence ++ stream index (i + 2) !
		outputSequence ++ (stream index (i + 1)) * 2 !
		outputSequence ++ ((stream index i) * 3) - 1) !
	}
}

OUTPUT ( outputSequence ) !
