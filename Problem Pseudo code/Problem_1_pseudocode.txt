--PROGRAM PSEUDOCODE PROBLEM 1

List sequenceOne = file.readline !
List sequenceTwo = file.readline !

List outputSequence !
i = 0 !

while (  i < (length(sequenceOne) div 2) - 1 ) {
outputSequence ++ sequenceOne index (i*2) !
outputSequence ++ sequenceOne index (i*2 + 1) !
outputSequence ++ sequenceTwo index i !
}

OUTPUT ( outputSequence ) !
