--PROGRAM PSEUDOCODE PROBLEM 1


-- Stream 
-- StreamVar = number of streams

List sequenceOne !
List sequenceTwo !

a = 0 !
while ( not (a == 0) ) then {
if ( ( a mod 2 ) == 0 ) then {
sequenceOne ++ stream index a !
}
if ( ( a mod 2 ) == 1 ) then {
sequenceTwo ++ stream index a !
}
a = a + 1 !
}

List outputSequence !
i = 0 !

while (  i < (length(sequenceOne) div 2) - 1 ) then {
outputSequence ++ sequenceOne index (i * 2) !
outputSequence ++ sequenceOne index (i * 2 + 1) !
outputSequence ++ sequenceTwo index i !
i = i + 1 !
}

OUTPUT ( outputSequence ) !
