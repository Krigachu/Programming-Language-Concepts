--PROGRAM PSEUDOCODE PROBLEM 1

List sequenceOne = file.readline¬

List outputSequence = []¬
Num countNum = 0¬

for (i = 0 to length(sequenceOne)) {
	countNum = 0¬
	for (j = 0 to i) {
		countNum = countNum + sequenceOne[j]¬
	}
	outputSequence = outputSequence ++ [countNum]¬
}

OUTPUT outputSequence¬
